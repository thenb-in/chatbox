@nt/chatbox — basic example

The chatbox is mounted in the bottom-right. It auto-expands a few seconds after page load. Click to launch as a centered modal. 

Chatbox.init({
  brand: { name: 'Acme Co', avatarText: 'AC', tagline: 'We reply in ~5 min' },
  steps: ['phone', 'interests', 'mail', 'name'],
  interests: [
    { key: 'sales',   label: 'Sales enquiry' },
    { key: 'support', label: 'Support' },
    { key: 'partner', label: 'Partnership' }
  ],
  whatsappNumber: '15555550100',
  autoExpandDelayMs: 4000,
  onSubmit: function (stage, payload) {
    console.log('chatbox submit', stage.stageName, payload.data);
  },
  onEvent: function (name, payload) {
    console.log('chatbox event', name, payload);
  }
});
